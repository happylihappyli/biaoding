using Common_Robot2;

namespace Test1
{
    public class S_Save_List : C_Node
    {

        public string file = ",";
        public string key_list = ",";

        public S_Save_List(string Name, C_Space space_parent, C_Space space) :
            base(space_parent, space)
        {
            this.Name = Name;
            space.vars_step.Add(Name, this);
        }


        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            List<string> pList = (List<string>)space.read_vars(pTrain, this, key_list, "List<string>");


            File.WriteAllLines(this.file, pList);

        }
    }
}
