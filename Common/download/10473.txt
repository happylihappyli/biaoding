
using Common_Robot2;

namespace Test1
{

    /// <summary>
    /// 文件名扩展名
    /// </summary>
    public class String_File_Extension : C_Node
    {
        public string key_read = "";
        public string key_save = "";

        public String_File_Extension(string name, C_Space space_parent, C_Space space) : 
            base(name,space_parent, space)
        {

        }


        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {

            string path = this.read_string("@"+this.key_read);
            string result=Path.GetExtension(path);

            this.save_var(this.key_save, "string", result);
        }

        public override void init()
        {
        }
    }
}
