using Common_Robot2;
using ConverxHull;
using System.Text;

namespace Test1
{
    public class C_Cloud_Read : C_Node
    {
        public string file = "";
        public string key_save = "cloud1";
        public string seperate = ",";

        public C_Cloud_Read(string name, C_Space space_parent, C_Space space) : base(name,space_parent, space)
        {
        }

        public override void init()
        {
            
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        private void run_sub_main()
        {
            string file_new =this.read_string(file);

            List<C_Point3D> list = Main.cloud_read(this,file_new, seperate);

            this.save_var(this.key_save, "List<C_Point3D>", list);

        }

    }
