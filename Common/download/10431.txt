
using Common_Robot2;

namespace Test1
{
    public class U_List_View_Selected_Index : C_Node
    {
        public ListView ui_item = null;

        public string key_ui = "";
        public string key_save = "";

        public U_List_View_Selected_Index(string name, C_Space space_parent, C_Space space) :
            base(name, space_parent, space)
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }


        private void run_sub_main()
        {

            S_List_View pItem = (S_List_View)space.vars_ui[key_ui];
            ui_item = pItem.ui_item;

            Action act = delegate ()
            {
                if (ui_item != null)
                {
                    if (ui_item.SelectedItems.Count > 0)
                    {
                        List<string> list1=new List<string>();  
                        
                        for (var i = 0; i < ui_item.SelectedItems.Count; i++)
                        {
                            ListViewItem item = ui_item.SelectedItems[i];
                            list1.Add(item.Index+"");
                        }
                        this.save_var(this.key_save, "list<string>", list1);
                    }
                }
            };
            if (ui_item != null)
                ui_item.BeginInvoke(act, null);
        }

        public override void init()
        {
            
        }
    }
}
