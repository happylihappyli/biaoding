using Common_Robot2;
using ConverxHull;

namespace Test1
{
    public class Data_Get_Group : C_Node
    {
        public string key_read = "cloud1";
        public string key_save = "min";

        public Data_Get_Group(string name, C_Space space_parent, C_Space space) : base(name,space_parent, space)
        {
        }


        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            int? group_id =null;

            if (this.key_read == "%main.group_id")
            {
                C_Data? pData = (C_Data?)this.read_var("main", "C_Data");
                group_id = pData?.Group_ID;
            }
            else
            {
                group_id = null;// (int?)this.read_var(this.key_read, "int");
            }

            this.save_var(this.key_save, "int", group_id);
        }


        public override void init()
        {
            
        }
    }
}
