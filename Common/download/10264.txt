using System.Collections.Generic;

namespace Test1
{
    public class C_Block_Complex
    {
        public List<C_Block_Simple> blocks=new List<C_Block_Simple>();

        public C_Box_Type pBox_Type;

        public int x_len = 0;
        public int y_len = 0;
        public int z_len = 0;
        public int x = 0;
        public int y = 0;
        public int z = 0;
        public bool 用完=false;

        public C_Block_Complex(
            C_Box_Type box_Type,
            int x_len, int y_len, int z_len)
        {
            this.pBox_Type = box_Type;
            this.x_len = x_len;
            this.y_len = y_len;
            this.z_len = z_len;
        }

        public int count()
        {
            int block_count = 0;
            for(var i = 0; i < blocks.Count; i++)
            {
                block_count += blocks[i].count;
            }
            return block_count;
        }


        public int space()
        {
            return this.x_len*this.y_len*this.z_len;
        }

        public double space_used()
        {
            double used = 0;
            for(var i = 0; i < blocks.Count; i++)
            {
                used+=blocks[i].space_used();
            }
            return used;
        }

        public C_Block_Complex Clone(C_Box_Type box_type)
        {
            C_Block_Complex p = new C_Block_Complex(box_type, x_len, y_len, z_len);
            for (var i = 0; i < blocks.Count; i++)
            {
                p.blocks.Add(blocks[i].Clone(box_type));  
            }
            return p;
        }
    }
}
