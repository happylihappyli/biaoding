using Common_Robot2;
using ConverxHull;

namespace Test1
{
    public class S_Var_Remove : C_Node
    {

        public string key_save = "var1";
        public string type = "string";

        public S_Var_Remove(string name, C_Space space_parent, C_Space space) :
            base(name, space_parent, space)
        {
        }

        public override void init()
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {

            if (pTrain == null) return;

            {
                space.remove_var(pTrain, this, key_save, type);
            }

        }
    }
}
