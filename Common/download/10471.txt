
using Common_Robot2;

namespace Test1
{
    public class String_SubStr : C_Node
    {
        public string key_read = "";
        public string key_save = "";
        public string index = "";
        public string len = "";

        public String_SubStr(string name, C_Space space_parent, C_Space space) : 
            base(name,space_parent, space)
        {
        }


        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {

            string line = this.read_string("@"+this.key_read);
            int i_index = this.read_int(index);
            string result = line.Substring(i_index);

            if (this.len != "")
            {
                string str_len = this.read_string(len);
                int i_len=int.Parse(str_len);
                result = line.Substring(i_index, i_len);
            }

            this.save_var(this.key_save, "string", result);
        }

        public override void init()
        {
        }
    }
}
