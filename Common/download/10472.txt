
using Common_Robot2;

namespace Test1
{
    /// <summary>
    /// 字符串长度
    /// </summary>
    public class String_Len : C_Node
    {
        public string key_read = "";
        public string key_save = "";

        public String_Len(string name, C_Space space_parent, C_Space space) : 
            base(name,space_parent, space)
        {
        }


        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            string line = this.read_string("@"+this.key_read);

            this.save_var(this.key_save, "string", line.Length+"");
        }

        public override void init()
        {
        }
    }
}
