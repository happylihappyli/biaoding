using Common_Robot2;
using System.Drawing.Imaging;
using static System.Windows.Forms.LinkLabel;


namespace Test1
{
    public class F_Show_Frm: C_Node
    {
        public string frm_name = "";
        public F_Show_Frm(string name, C_Space space_parent, C_Space space) :
            base(name,space_parent, space)
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            Type type = Type.GetType("Test1."+ frm_name);
            Form thisFrm = (Form)Activator.CreateInstance(type);
            thisFrm.ShowDialog();
        }


        public override void init()
        {
        }
    }
}
