
using Common_Robot2;

namespace Test1
{
    public class U_List_View_Read_Item : C_Node
    {
        public ListView ui_item = null;

        public string key_ui = "";
        public string index = "-1";
        public string split = ",";
        public string key_save = "";

        public U_List_View_Read_Item(string name, C_Space space_parent, C_Space space) :
            base(name,space_parent, space)
        {
        }
        public override void init()
        {

        }
        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }


        private void run_sub_main()
        {
            S_List_View? pItem = null;
            if (space.vars_ui.ContainsKey(key_ui))
            {
                pItem = (S_List_View?)space.vars_ui[key_ui];
            }
            else if (space.parent.vars_ui.ContainsKey(key_ui))
            {
                pItem = (S_List_View?)space_parent.vars_ui[key_ui];
            }
            else if (space.parent.parent.vars_ui.ContainsKey(key_ui))
            {
                pItem = (S_List_View?)space.parent.parent.vars_ui[key_ui];
            }
            else
            {
                MessageBox.Show(this.Name + "UI没有这个控件");
            }

            //S_List_View pItem = (S_List_View)space.vars_ui[key_ui];
            ui_item = pItem.ui_item;

            int icount = 0;
            Action act = delegate ()
            {
                if (ui_item != null)
                {
                    if (ui_item.SelectedItems.Count > 0)
                    {
                        ListViewItem item = ui_item.SelectedItems[0];
                        string line = item.Text;

                        if (this.key_save.IndexOf(",") > -1)
                        {
                            string[] strSplit= key_save.Split(',');
                            this.save_var(strSplit[0], "string", line);

                            for (var i = 1; i < strSplit.Length; i++)
                            {
                                if (i<item.SubItems.Count)
                                {
                                    line = item.SubItems[i].Text;
                                    this.save_var(strSplit[i], "string", line);
                                }
                                else
                                {
                                    this.save_var(strSplit[i], "string", "");
                                }
                            }
                        }
                        else
                        {
                            for (var i = 1; i < item.SubItems.Count; i++)
                            {
                                line += split + item.SubItems[i].Text;
                            }
                            this.save_var(this.key_save, "string", line);
                        }
                        icount = 10000;
                    }
                }
            };
            if (ui_item != null)
                ui_item.BeginInvoke(act, null);

            while (icount<1000)
            {
                icount++;
                Thread.Sleep(10);
            }
        }
    }
}
