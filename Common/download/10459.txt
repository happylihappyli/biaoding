
using System.Security.Cryptography;
using System.Text;
using Common_Robot2;

namespace Test1
{
    public class String_MD5 : C_Node
    {
        public string key_read = "";
        public string key_save = "";
        public String_MD5(string name, C_Space space_parent, C_Space space) : base(name,space_parent, space)
        {
        }


        public string GetMd5Hash(string input)
        {
            using (MD5 md5 = MD5.Create())
            {
                // Convert the input string to a byte array
                byte[] inputBytes = Encoding.UTF8.GetBytes(input);

                // Compute the hash
                byte[] hashBytes = md5.ComputeHash(inputBytes);

                // Convert the hash bytes to a hexadecimal string
                StringBuilder sb = new StringBuilder();
                for (int i = 0; i < hashBytes.Length; i++)
                {
                    sb.Append(hashBytes[i].ToString("x2"));
                }

                return sb.ToString().ToLower();
            }
        }


        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {

            string line = (string)this.read_var(this.key_read,"string");

            string line2= GetMd5Hash(line);

            this.save_var(this.key_save, "string", line2);
        }

        public override void init()
        {
        }
    }
}
