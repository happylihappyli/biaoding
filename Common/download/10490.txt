
using Common_Robot2;

namespace Test1
{
    public class File_Last_Time: C_Node
    {
        public string file = "";
        public string key_save = "";

        public File_Last_Time(string name, C_Space space_parent, C_Space space) 
            : base(name,space_parent, space)
        {
        }



        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            string str_path = this.read_string(file);

            if (File.Exists(str_path) == false)
            {
                this.save_var(this.key_save, "string", "");
                return;
            }
            this.save_var(this.key_save, "string", "");

        }

        public override void init()
        {
        }
    }

}
