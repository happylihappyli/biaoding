using Common_Robot2;
using System.IO;

namespace Test1
{
    public class File_Dir_Files: C_Node
    {
        public string path = "";
        public string key_save ="";
        public string filter = "";
        public string type = "0";//是否显示文件类型，如果=1就显示

        public File_Dir_Files(string name, C_Space space_parent, C_Space space) 
            : base(name,space_parent, space)
        {
        }



        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            string path2 = this.read_string(path);
            if (path2.EndsWith("\\")==false)
            {
                path2 += "\\";
            }

            if (Directory.Exists(path2)==false)
            {
                return;
            }

            List<string> pList = new List<string>();
            string[] fileEntries =  Directory.GetFiles(path2);
            foreach (string fileName in fileEntries)
            {
                if (fileName.EndsWith(this.filter))
                {
                    FileInfo fileInfo = new FileInfo(fileName);
                    //if (type == "1")
                    //{

                    //    pList.Add(fileName.Substring(path2.Length) + "|f||"+ fileInfo.LastWriteTime);
                    //}
                    //else
                    {
                        pList.Add(fileName.Substring(path2.Length) + "|f||" + fileInfo.LastWriteTime);
                    }
                }
            }
            this.save_var(key_save, "List<string>", pList);

        }
        public override void init()
        {

        }

    }

}
