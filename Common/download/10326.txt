using Common_Robot2;
using ConverxHull;

namespace Test1
{
    public class F_Call : C_Node
    {
        public string key_read = "";
        public string str_new = "0";
        public string wait = "1000";

        public F_Call(string name, C_Space space_parent, C_Space space) 
            : base(name,space_parent, space)
        {
        }

        public override void init()
        {
            
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            if (wait != "")
            {
                Thread.Sleep(int.Parse(wait));
            }
            if (space.vars_step.ContainsKey(key_read) == false)
            {
                return;
            }
            C_Node pStep = space.vars_step[key_read];
            if (str_new == "1")
            {

                Random rnd = new Random();
                int train_id = rnd.Next(100, 900);
                space.pTrain = CommonMain.create_train("Mouse_" + train_id);

                pStep.Run(space.pTrain);
            }
            else
            {
                pStep.Run(pTrain);
            }
        }
    }
}
