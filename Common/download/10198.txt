using Common_Robot2;

namespace Test1
{
    public class Img_Save : C_Node
    {
        public string key_read = "2d_pic";
        public string key_save = "";
        public string key_time_id = "";
        public string file = "";

        public Img_Save(string name, C_Space space_parent, C_Space space) :
            base(name, space_parent, space)
        {
        }

        public override void init()
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }


        public void run_sub_main()
        {
            Bitmap? bmp = (Bitmap?)this.read_var(this.key_read, "Bitmap");
            string file2 = this.read_string(this.file);
            file2 = Main.file_replace(pTrain, file2);
            this.save_var(this.key_save, "string", file2);
            
            try
            {
                Bitmap bmp2 = Main.CopyBmp(bmp);
                lock (bmp2)
                {
                    string path=Path.GetDirectoryName(file2);
                    if (Directory.Exists(path)==false)
                    {
                        Directory.CreateDirectory(path);
                    }
                    bmp2.Save(file2);
                }
            }
            catch(Exception ex)
            {
                Main.WriteLine(this,ex.ToString());
            }
        }


    }
}
