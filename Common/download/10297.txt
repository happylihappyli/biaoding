using Common_Robot2;

namespace Test1
{
    public class S_Cmd_Add : C_Node
    {
        public string key_save = "";
        public string command = "";

        public S_Cmd_Add(string name, C_Space space_parent, C_Space space) :
            base(name,space_parent, space)
        {
        }

        public override void init()
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        private void run_sub_main()
        {
            List<string>? pList = null;
            if (space.contains(pTrain, this, this.key_save,"List<string>"))
            {
                pList = (List<string>?)this.read_var(this.key_save, "");
            }
            else
            {
                pList = new List<string>();
                this.save_var(this.key_save, "List<string>", pList);
            }

            string lines = this.read_string(command);
            lines = lines.Replace("\r\n", "\n");
            lines = lines.Replace("\r", "\n");

            string[] strSplit = lines.Split("\n");
            for(var i=0;i<strSplit.Length;i++)
            {
                string line = strSplit[i].Trim();
                if (line != "")
                {
                    pList?.Add(line);
                }
            }
        }

    }

    
}
