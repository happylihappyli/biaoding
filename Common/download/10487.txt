using Common_Robot2;
using Funny;

namespace Test1
{
    public class String_AES_Encrypt : C_Node
    {
        public string read  = "";
        public string key_save = "";
        public string password = "";

        public String_AES_Encrypt(string name, C_Space space_parent, C_Space space) :
            base(name,space_parent, space)
        {
        }

        public override void init()
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        private void run_sub_main()
        {

            string value = this.read_string(this.read);
            string password = this.read_string(this.password);

            c_string pString = new c_string();
            if (password == "")
            {
                password = pString.default_password;
            }
            var str_password = pString.aes_encrypt(value, password);

            this.save_var(key_save, "string", str_password);
        }
    }

    
}
