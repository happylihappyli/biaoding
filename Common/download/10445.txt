using Common_Robot2;
using ConverxHull;

namespace Test1
{
    public class C_Cloud_From_Picture_Polygon : C_Node
    {
        public string key_ui = "";
        public string key_save = "cloud1";

        public C_Cloud_From_Picture_Polygon(string name, C_Space space_parent, C_Space space) : base(name,space_parent, space)
        {

        }



        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            S_PictureBox? pItem;
            if (space.vars_ui.ContainsKey(key_ui))
            {
                pItem = (S_PictureBox)space.vars_ui[key_ui];
            }
            else
            {
                pItem = (S_PictureBox)space_parent.vars_ui[key_ui];
            }

            if (pItem == null)
                return;

            this.save_var(this.key_save, "List<C_Point3D>", pItem.polygonPoints);

        }

        public override void init()
        {
        }
    }
}
