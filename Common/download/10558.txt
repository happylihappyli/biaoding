using Common_Robot2;

namespace Test1
{
    public class U_List_View_To_List : C_Node
    {
        public ListView ui_item = null;

        public string key_ui = "";
        public string key_save = "";
        public string index = "0";
        public string selected = "0";
        public string sep = "|";

        public U_List_View_To_List(string name, C_Space space_parent, C_Space space) :
            base(name,space_parent, space)
        {
        }

        public override void init()
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        private void run_sub_main()
        {

            S_List_View pItem = (S_List_View)space.vars_ui[key_ui];
            ui_item = pItem.ui_item;

            int i_index = int.Parse(this.read_string(index));

            Action act = delegate ()
            {
                if (ui_item != null)
                {
                    List<string> list = new List<string>();
                    if (this.selected == "1")
                    {

                        for (var k = 0; k < ui_item.SelectedItems.Count; k++)
                        {
                            ListViewItem item = ui_item.SelectedItems[k];
                            if (i_index == -1)
                            {
                                string file = "";
                                for (var j = 0; j < item.SubItems.Count; j++)
                                {
                                    file += item.SubItems[j].Text+sep;
                                }

                                list.Add(file);

                            }
                            else
                            {

                                string file = item.SubItems[i_index].Text;

                                list.Add(file);
                            }
                        }
                    }
                    else{

                        for (var k = 0; k < ui_item.Items.Count; k++)
                        {
                            ListViewItem item = ui_item.Items[k];
                            string file = item.SubItems[i_index].Text;

                            list.Add(file);
                        }
                    }
                    this.save_var(this.key_save, "List<string>", list);
                }
                //call_next(pTrain);
            };
            if (ui_item != null)
                ui_item.BeginInvoke(act, null);
        }
    }
}
