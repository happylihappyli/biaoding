
using Common_Robot2;

namespace Test1
{
    public class Img_Read : C_Node
    {
        public string? file="";
        public string? key_read="";

        public Img_Read(string name, C_Space space_parent, C_Space space) :
            base(name,space_parent, space)
        {
        }

        public override void init()
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            string str_file = this.read_string(this.file);
            if (str_file== null || str_file=="")
            {
                S_TTS.speak_async("文件名为空");
                return;
            }
            try
            {
                Bitmap bmp1 = new Bitmap(str_file);
                this.save_var(this.key_read, "Bitmap", bmp1);
            }
            catch(Exception ex)
            {
                MessageBox.Show(ex.ToString());
            }
            

        }

    }
}
