
using Common_Robot2;
using ConverxHull;

namespace Test1
{
    public class C_Cloud_To_List_String : C_Node
    {
        public string key_read = "cloud1";
        public string key_save = "file_name";
        public string template = "";

        public C_Cloud_To_List_String(string name, C_Space space_parent, C_Space space) :
            base(name,space_parent, space)
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;

        }


        private void run_sub_main()
        {

            List<C_Point3D>? pList = (List<C_Point3D>?)this.read_var(this.key_read, "List<C_Point3D>");
            if (pList == null)
            {
                MessageBox.Show(this.Name + "  key_read ==null ");
                return;
            }

            List<string> pList2 = new List<string>();

            for(var i=0;i<pList.Count;i++)
            {
                C_Point3D p = pList[i];
                string line = template;
                line = line.Replace("{id}", i+"");
                line = line.Replace("{x}", p.x.ToString());
                line = line.Replace("{y}", p.y.ToString());
                line = line.Replace("{z}", p.z.ToString());
                pList2.Add(line);
            }

            this.save_var(this.key_save, "List<string>", pList2);

        }

        public override void init()
        {
            
        }
    }
}
