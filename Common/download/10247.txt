
using Common_Robot2;

namespace Test1
{
    public class Cal_Math_Round : C_Node
    {

        public string key_read = "var1";
        public string key_save = "var1";
        public string count = "0";//保留小数个数


        public Cal_Math_Round(string name, C_Space space_parent,C_Space space) : base(name,space_parent, space)
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            object obj= this.read_var(key_read, "");
            double db1 = Main.get_double_from_obj(this,obj);

            string result = Math.Round(db1,int.Parse(count)).ToString();
            
            this.save_var(key_save,"string", result);
        }

        public override void init()
        {
            this.init_setting();
        }
    }
}
