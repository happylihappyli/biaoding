
using Common_Robot2;
using System.Drawing.Imaging;


namespace Test1
{
    //"下取整
    public class Cal_Floor: C_Node
    {
        //"读取的数字变量"
        public string key_read = "";
        //保存的数字变量
        public string key_save = "";


        public Cal_Floor(string name, C_Space space_parent, C_Space space) :
            base(name,space_parent, space)
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            string num = this.read_string("@" + this.key_read);
            double db_num = double.Parse(num);
            this.save_var(this.key_save, "string", Math.Floor(db_num));
        }

        public override void init()
        {

        }
    }
}
