using Common_Robot2;
using ConverxHull;

namespace Test1
{
    public class C_Cloud_Append : C_Node
    {
        public string key_read= "cloud1";
        public string key_read_point = "point";
        public string key_save = "cloud2";

        public C_Cloud_Append(string name, C_Space space_parent, C_Space space) : base(name,space_parent, space)
        {
        }



        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            List<C_Point3D>? pList = (List<C_Point3D>?)this.read_var(this.key_read, "List<C_Point3D>");

            if (pList == null)
            {
                MessageBox.Show(this.Name + "key_read 读取为空");
                return;
            } 

            List<C_Point3D> pList2 = new List<C_Point3D>();

            C_Point3D? pPointNew = (C_Point3D?)this.read_var(this.key_read_point, "List<C_Point3D>"); //space.vars.read_vars(pTrain, this, this.key_read_point, "C_Point3D");

            if (pList == null || pPointNew==null)
            {
                MessageBox.Show(this.Name + "key_read_point 读取为空");
                return;
            }


            for (var i = 0; i < pList.Count; i++)
            {
                C_Point3D p = pList[i];
                pList2.Add(p.Clone());
            }
            pList2.Add(pPointNew);

            this.save_var(this.key_save, "List<C_Point3D>", pList2);

        }

        public override void init()
        {
        }
    }
}
