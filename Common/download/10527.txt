
using Common_Robot2;

namespace Test1
{
    public class File_From_Base64 : C_Node
    {
        public string key_read = "";
        public string file = "";

        public File_From_Base64(string name, C_Space space_parent, C_Space space)
            : base(name, space_parent,space)
        { 
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            string code = this.read_string("@"+this.key_read);
            byte[] bs= Convert.FromBase64String(code);

            string str_file = this.read_string(this.file);
            File.WriteAllBytes(str_file, bs);
        }

        public override void init()
        {
        }
    }
}
