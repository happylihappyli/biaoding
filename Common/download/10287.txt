
using Common_Robot2;

namespace Test1
{
    public class File_To_Base64 : C_Node
    {
        public string file = "";
        public string key_save = "";

        public File_To_Base64(string name, C_Space space_parent, C_Space space) 
            : base(name,space_parent, space)
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            string file = this.read_string(this.file);
            string line2 = Main.FileToBase64Str(file);
            this.save_var(key_save, "string", line2);
        }


        public override void init()
        {
        }
    }
}
