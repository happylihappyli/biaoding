
using Common_Robot2;
using System.Text;

namespace Test1
{
    public class File_Save: C_Node
    {
        public string file = "";
        public string encode = "utf-8";
        public string key_read ="";
        public string key_time_id = "";

        public File_Save(string name, C_Space space_parent,C_Space space) : base(name, space_parent,space)
        {
        }

        public override void init()
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            if (this.encode == "")
            {
                this.encode = "utf-8";
            }

            lock (this.file)
            {
                string file_save =this.read_string(this.file);
                string time_id = this.read_string(this.key_time_id);

                file_save = file_save.Replace("@", time_id);

                string? path_dir=Path.GetDirectoryName(file_save);
                if (Directory.Exists(path_dir)==false)
                {
                    Directory.CreateDirectory(path_dir);
                }

                FileStream fs_vector = new FileStream(file_save, FileMode.Create, FileAccess.Write);
                StreamWriter w_vector = new StreamWriter(fs_vector, Encoding.GetEncoding(this.encode));// Encoding.UTF8); ;

                string? strLine = (string?)this.read_var(this.key_read,"string");
                w_vector.Write(strLine);

                w_vector.Close();
                fs_vector.Close();
            }
                
        }
    }

}
