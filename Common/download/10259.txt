namespace Test1
{
    public class String_JArray_To_List_String : C_Node
    {

        public string key_read = "";//string
        public string key_save = "";//C_Point3D

        public String_JArray_To_List_String(string Name, C_Space space_parent, C_Space space) : base(space_parent,space)
        {
            this.Name = Name;
            space.vars_step.Add(Name, this);
        }


        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            JArray arr = (JArray)this.read_var(this.key_read, "JArray");

            List<string> list = new List<string>();
            for (var i = 0; i < arr.Count; i++)
            {
                string line = arr[i].ToString();
                list.Add(line);
            }
            space.save_vars(pTrain, this,this.key_save, "List<string>", list);

        }

        public void init()
        {

        }
    }
}
