
using Common_Robot2;
using ConverxHull;

namespace Test1
{
    public class S_Point_xyz : C_Node
    {
        public string key_read = "10";
        public string type="x";
        public string key_save = "";

        public S_Point_xyz(string name, C_Space space_parent, C_Space space) : 
            base(name,space_parent, space)
        {
        }

        public override void init()
        {
            
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            C_Point3D? pPoint = (C_Point3D?)this.read_var(key_read, "C_Point3D");
            if (pPoint == null)
            {
                Main.WriteLine(this,"Error key_read=" + this.key_read + " 读取为空！");
                this.Next_Step = Node_Next.False;
                return;
            }

            double value = 0;
            switch (this.type)
            {
                case "x":
                    value=pPoint.x;
                    break;
                case "y":
                    value = pPoint.y;
                    break;
                case "z":
                    value = pPoint.z;
                    break;
            }


            Main.WriteLine(this, this.key_save+"="+value);

            this.save_var(this.key_save, "string", value+"");
            this.Next_Step = Node_Next.True;
        }

    }



}
