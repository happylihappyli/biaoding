
using Common_Robot2;

namespace Test1
{
    public class S_CheckBox : S_UI
    {
        public CheckBox? pControl;
        public string text = "选择";
        public string read_key = "";

        public S_CheckBox(string Name, C_Space space_parent, C_Space space):
            base(Name, space_parent, space)
        {
            this.Name = Name; 
        }

        public void set_checked(bool value)
        {
            if (pControl!= null)
            {
                pControl.Checked = value;
            }
        }

        public void init()
        {
            space.vars_ui.TryAdd(this.key, this);

            pControl = new CheckBox();
            pControl.Location = new System.Drawing.Point(x, y);
            pControl.Size = new Size(width, height);
            
            var font_name = this.font_name;
            if (font_name == "")
            {
                font_name = "宋体";
            }
            var font_size = this.font_size;
            if (this.font_size == "")
            {
                font_size = "12";
            }
            pControl.Font = new Font(font_name, float.Parse(font_size),
                FontStyle.Regular, GraphicsUnit.Point, ((byte)(134)));
            pControl.Text = text;


            if (text.StartsWith("@#"))
            {
                this.key_field = text.Substring(1);
                space.pUI_Update.TryAdd(this.key_field,this);
            }


            pControl.TextChanged += Text_Change;

            if (this.color_bg != "")
            {
                pControl.BackColor = ColorTranslator.FromHtml(this.color_bg);
            }
            if (this.color_font != "")
            {
                pControl.ForeColor = ColorTranslator.FromHtml(this.color_font);
            }


            Action act = delegate ()
            {
                if (pFrm != null)
                {
                    Main.Add_Panel(this.pFrm, dock, this.group, space, pControl);
                }
            };
            if (pFrm != null)
                pFrm.BeginInvoke(act, null);


            this.event_change_value += S_TextBox_event_change_value;
        }

        private void S_TextBox_event_change_value()
        {

            Action act = delegate ()
            {
                if (pFrm != null)
                {
                    pControl.Text = (string)this.data;
                }
            };
            if (pFrm != null)
                pFrm.BeginInvoke(act, null);
        }

        private void Text_Change(object sender, EventArgs e)
        {
            if (this.key_field != "")
            {
                this.save_var(this.key_field, "string", pControl.Text);
            }
        }

        public override Task run_sub()
        {
            run_sub_main();
             
            return Task.CompletedTask;
        }

        public void run_sub_main()
        {
            Main.WriteLine(this.Name + " run");

            string value = "";
            if (pTrain.get_Vars().ContainsKey(read_key))
            {
                var tmp = pTrain.get_Vars()[read_key].pObj;
                if (tmp.GetType() == typeof(string))
                {
                    value = (string)tmp;
                }
                else
                {
                    value = tmp.ToString();
                }
            }

            Action act = delegate ()
            {
                if (pControl != null)
                {
                    pControl.Text = value;
                }
            };
            if (pControl != null)
            {
                pControl.BeginInvoke(act, null);
            }
        }
    }
}
