using Common_Robot2;
using Newtonsoft.Json.Linq;

namespace Test1
{
    public class U_List_View_From_Json : C_Node
    {
        public ListView ui_item = null;

        public string key_ui = "";
        public string key_json = "";
        public string fields="a,b";

        public U_List_View_From_Json(string name, C_Space space_parent, C_Space space) :
            base(name,space_parent, space)
        {
        }

        public override void init()
        {
        }

        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        private void run_sub_main()
        {

            S_List_View pItem = (S_List_View)space.vars_ui[key_ui];
            ui_item = pItem.ui_item;


            Action act = delegate ()
            {
                if (ui_item != null)
                {
                    ui_item.Items.Clear();

                    string? lines = (string?)this.read_var(this.key_json, "string");
                    try
                    {

                        JArray arr = JArray.Parse(lines);

                        if (arr != null)
                        {
                            for (var k = 0; k < arr.Count; k++)
                            {
                                JObject item = (JObject)arr[k];
                                if (item == null) continue;

                                string[] strSplit = fields.Split(",");
                                string key = strSplit[0];
                                if (key == null || key == "") continue;

                                var obj = item[key];
                                if (obj == null) continue;

                                string name = obj.ToString();
                                ListViewItem item2 = ui_item.Items.Add(name);

                                for (var i = 1; i < strSplit.Length; i++)
                                {
                                    key = strSplit[i];
                                    obj = item[key];
                                    if (obj == null) continue;
                                    string value = obj.ToString();
                                    item2.SubItems.Add(value);
                                }
                            }
                        }
                    }
                    catch (Exception ex)
                    {
                        MessageBox.Show(lines);
                    }
                }
            };
            if (ui_item != null)
                ui_item.BeginInvoke(act, null);
        }
    }
}
