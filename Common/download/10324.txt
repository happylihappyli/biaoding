using Common_Robot2;

namespace Test1
{
    public class String_Split : C_Node
    {
        public string key_read = "";
        public string key_save = "";
        public string split = ",";
        public String_Split(string name, C_Space space_parent, C_Space space) : 
            base(name,space_parent, space)
        {
        }
        public override void init()
        {

        }
        public override Task run_sub()
        {
            run_sub_main();
            return Task.CompletedTask;
        }

        private void run_sub_main()
        {
            string value = (string)space.read_vars(pTrain, this, this.key_read, "string");
            if (value == null) return;

            List<string> list = new List<string>();

            split = split.Replace("\\r", "\r");
            split = split.Replace("\\n", "\n");


            string[] strSplit = value.Split(split);
            for (var i = 0; i < strSplit.Length; i++)
            {
                list.Add(strSplit[i]);
            }
            space.save_vars(pTrain, this, key_save, "List<string>", list);

        }
    }
}
